 41 class StackOfInteger {
 42 
 43     private int[] elements;
 44     private int size;
 45     public static final int DEFAULT_CAPACITY = 16;
 46 
 47     public StackOfInteger() {
 48         this(DEFAULT_CAPACITY);

 51 
 52     public StackOfInteger(int capacity) {
 53         elements = new int[capacity];
 54     }
 55 
 56     // test if the stack is empty 
 57     public boolean empty() {
 58         return size == 0;
 59     }
 60 
 61     /*A method push(integer value) that stores value at the top of the stack. 
 62 This method checks if the stack is full before pushing to it. In case the stack is full, it
 63 i. copies the contents of the stack to a new temp array
 64 ii. creates a new elements array with double the current capacity
 65 iii. copies back temp to elements
 66 iv. places value at the top of the new stack */
 67 
 68    public void push(int value) {
 69         if (size >= elements.length) {
 70             int[] temp = new int[elements.length * 2];
 71             System.arraycopy(temp, 0, temp, 0, elements.length);
 72             elements = temp;
 73         }
 74         elements[size++] = value;
 75     }
 76 
 77     public int pop() {
 78         if (this.empty()) {
 79             System.out.println("Stack is empty");
 80             return 0;
 81 
 82         } else {
 83             return elements[--size];
 84         }
 85     }
 86     
 87     public int peek(){
 88         if (this.empty()) {
 89             System.out.println("Stack is empty");
 90             return 0;
 91 
 92         } else {
 93             return elements[size - 1];
 94         }
 95         
 96     }
 97     
 98     public void printStack() {
 99         System.out.print("Stack ( top to bottom ): ");
100         for (int i = size -1; i> -1; i--){
101             System.out.print(elements[i] + " \n");
102         }
